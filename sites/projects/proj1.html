Работа с ультрозвуковым датчиком HC-SR04

@ISAAC100

В этом проекте я познакомлю вас с ультразвуковым датчиком HC-SR04. Он работает, посылая звуковые волны от передатчика, которые затем отскакивают от объекта и затем возвращаются в приемник. Вы можете определить, как далеко что-то находится, к тому времени, которое требуется звуковым волнам, чтобы вернуться к датчику. Давайте получим право на это!

/*
 * HC-SR04 example sketch
 *
 * https://create.arduino.cc/projecthub/Isaac100/getting-started-with-the-hc-sr04-ultrasonic-sensor-036380
 *
 * by Isaac100
 */

const int trigPin = 9;
const int echoPin = 10;

float duration, distance;

void setup() {
  pinMode(trigPin, OUTPUT);
  pinMode(echoPin, INPUT);
  Serial.begin(9600);
}

void loop() {
  digitalWrite(trigPin, LOW);
  delayMicroseconds(2);
  digitalWrite(trigPin, HIGH);
  delayMicroseconds(10);
  digitalWrite(trigPin, LOW);

  duration = pulseIn(echoPin, HIGH);
  distance = (duration*.0343)/2;
  Serial.print("Distance: ");
  Serial.println(distance);
  delay(100);
}
